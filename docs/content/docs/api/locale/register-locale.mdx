Register a locale for use with fdu.

## Syntax

```typescript
registerLocale(name: string, config: LocaleConfig): void
```

## Parameters

- **name**: `string` - Unique identifier for the locale
- **config**: `LocaleConfig` - Locale configuration object
  - `months`: `string[]` - Full month names (12 items)
  - `monthsShort`: `string[]` - Short month names (12 items)
  - `weekdays`: `string[]` - Full weekday names (7 items, Sunday first)
  - `weekdaysShort`: `string[]` - Short weekday names (7 items)
  - `weekdaysMin`: `string[]` - Minimal weekday names (7 items)

## Examples

### Register Built-in Locale

```typescript
import { registerLocale } from "@pyyupsk/fdu";
import { es } from "@pyyupsk/fdu/locale/es";

registerLocale("es", es);
```

### Register Multiple Locales

```typescript
import { registerLocale } from "@pyyupsk/fdu";
import { es } from "@pyyupsk/fdu/locale/es";
import { fr } from "@pyyupsk/fdu/locale/fr";
import { de } from "@pyyupsk/fdu/locale/de";

registerLocale("es", es);
registerLocale("fr", fr);
registerLocale("de", de);
```

### Register Custom Locale

```typescript
import { registerLocale } from "@pyyupsk/fdu";

registerLocale("custom", {
  months: [
    "January", "February", "March", "April",
    "May", "June", "July", "August",
    "September", "October", "November", "December"
  ],
  monthsShort: [
    "Jan", "Feb", "Mar", "Apr", "May", "Jun",
    "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"
  ],
  weekdays: [
    "Sunday", "Monday", "Tuesday", "Wednesday",
    "Thursday", "Friday", "Saturday"
  ],
  weekdaysShort: [
    "Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"
  ],
  weekdaysMin: [
    "Su", "Mo", "Tu", "We", "Th", "Fr", "Sa"
  ]
});
```

### Override Existing Locale

```typescript
import { registerLocale } from "@pyyupsk/fdu";

// Register custom English variant
registerLocale("en-custom", {
  months: [
    "Januar", "Februar", "March", // ... custom names
  ],
  // ... other config
});
```

## Complete Example

```typescript
import { fdu, registerLocale, locale } from "@pyyupsk/fdu";
import { es } from "@pyyupsk/fdu/locale/es";
import { fr } from "@pyyupsk/fdu/locale/fr";

// Register locales
registerLocale("es", es);
registerLocale("fr", fr);

// Set global locale
locale("es");

// Use locales
const date = fdu("2025-01-15");

console.log(date.format("MMMM DD, YYYY"));
// 'enero 15, 2025'

console.log(date.locale("fr").format("DD MMMM YYYY"));
// '15 janvier 2025'
```
